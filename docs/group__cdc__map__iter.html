Parameters

Example:

struct cdc_map *map = NULL;
...
struct cdc_map_iter it = CDC_INIT_STRUCT;

// handle error
}
cdc_map_begin(map, &it);
while (cdc_map_iter_has_next(&it)) {
void *value = cdc_map_iter_value(&it);
...

}
